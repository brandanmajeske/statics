package
{
    
    import com.brandanmajeske.managers.SlideManager;
    import com.brandanmajeske.time.TimeZoneUtil;
    import com.brandanmajeske.converters.KilobyteConverterUtil;
    import com.brandanmajeske.converters.InchesToFeet;
    import com.brandanmajeske.ui.LayoutBox;
    import flash.display.Sprite;
    import flash.events.Event;
    import flash.events.MouseEvent;
    import flash.text.Font;
    import flash.text.TextField;
    import flash.text.TextFormat;
    import lib.SliderDisplay;
    
    [SWF("800", height="400", backgroundColor="#f5f5f5")]
    public class Main extends Sprite
    {
        // private variables
        private var _lb:LayoutBox;
        private var _sd:SliderDisplay;
        private var _sdm:SlideManager;
        private var _sm:SlideManager;
        private var _sdInches:TextField;
        private var _sdFeet:TextField;
        private var _hf:TextFormat; 
        private var _tf:TextFormat
        private var _ksd:SliderDisplay;
        private var _ksm:SlideManager;
        private var _kilobytes:TextField;
        private var _kbToMB:TextField;
        private var _mbToGB:TextField;
        private var _gbToTB:TextField;
        private var _tsd:SliderDisplay;
        private var _tsm:SlideManager;
        private var _pacific:TextField;
        private var _mountain:TextField;
        private var _central:TextField;
        private var _eastern:TextField;
        
        //constructor
        public function Main()
        {
            
            //layout box for the UIs to be displayed in
            _lb = new LayoutBox(20, true);
            this.addChild(_lb);
            _lb.x = 80;
    
            //text format for the UIs
            _hf = new TextFormat();
            _hf.size = 26;
            _hf.font = "Helvetica";
            _hf.color = 0x333333;
            //for text format 
            _tf = new TextFormat();
            _tf.size = 20;
            _tf.font = "Arial";
            
            //the functions for the various UIs called here
            inchesToFeetConverter();
            kilobyteConverter();
            timeZoneConverter();
            
        }
        
        // A Time Zone converter that calculates the various Main-land US Times Zones
        // using a static function in the TimeZoneUtil class that acts on a value passed to the function
        // The value is manipulated by a slider
        
        private function timeZoneConverter():void
        {
            // add the slider asset to the stage in the layoutbox
            _tsd = new SliderDisplay();
            _tsd.scaleX = _tsd.scaleY = .75;
            _lb.addChild(_tsd);
            _tsd.y = _ksd.y;
            
            //add header text
            var hdt:TextField = new TextField();
            hdt.text = "US Time Zones";
            hdt.x = 20;
            hdt.y = -150;
            hdt.width = 220;
            hdt.mouseEnabled = false;
            hdt.setTextFormat(_hf);
            _tsd.addChild(hdt);
            
            //add time zone text
            _pacific = new TextField();
            _pacific.mouseEnabled = false;
            _pacific.text = "Pacific:        12:00";
            _tsd.addChild(_pacific);
            _pacific.setTextFormat(_tf);
            _pacific.width = 250;
            _pacific.y = -120;
            
            //add time zone text
            _mountain = new TextField();
            _mountain.text = "Mountain:    1:00";
            _mountain.mouseEnabled = false;
            _tsd.addChild(_mountain);
            _mountain.setTextFormat(_tf);
            _mountain.width = 200;
            _mountain.y = -95;
            
            //add time zone text
            _central = new TextField();
            _central.mouseEnabled = false;
            _central.text = "Central:       2:00";
            _tsd.addChild(_central);
            _central.setTextFormat(_tf);
            _central.width = 200;
            _central.y = -70;
            
            
            //add time zone text
            _eastern = new TextField();
            _eastern.mouseEnabled = false;
            _eastern.text = "Eastern:      3:00";
            _tsd.addChild(_eastern);
            _eastern.setTextFormat(_tf);
            _eastern.width = 200;
            _eastern.y = -45;
            _pacific.x = _mountain.x = _central.x = _eastern.x = 25;
            
            //slider manager function and event listener for the handle position changing
            _tsm = new SlideManager();
            _tsm.setUpAssets(_tsd.mcTrack, _tsd.mcHandle);
            _tsm.addEventListener(Event.CHANGE, onHandlePositionChange);
            
        }
        
        // function that listens for the change to the posistion of the handle, then calls 
        // the static utility function to display time zone values in the appropriate fields
        
        private function onHandlePositionChange(event:Event):void
        {
            _tsm = SlideManager(event.currentTarget);
            if(_tsm.percent >= 0.995)
            {
                _tsm.percent = Math.ceil(_tsm.percent);    
            }
            
            _pacific.text = "Pacific:        "+TimeZoneUtil.pacificTimeZone(_tsm.percent)+":00";
            _mountain.text = "Mountain:    "+TimeZoneUtil.mountainTimeZone(_tsm.percent)+":00";
            _central.text = "Central:       "+TimeZoneUtil.centralTimeZone(_tsm.percent)+":00";
            _eastern.text = "Eastern:      "+TimeZoneUtil.easternTimeZone(_tsm.percent)+":00";
            
            //format the text fields 
            _pacific.setTextFormat(_tf);
            _mountain.setTextFormat(_tf);
            _central.setTextFormat(_tf);
            _eastern.setTextFormat(_tf);

        }
        
        // Byte conversion from Kilobytes up to Terabytes using
        // a slider and a statid utility function
        
        private function kilobyteConverter():void
        {
            // add the slider asset to the stage in the layoutbox
            _ksd = new SliderDisplay();
            _ksd.scaleX = _ksd.scaleY = .75;
            _lb.addChild(_ksd);
            _ksd.y = _sd.y;
            
            _ksm = new SlideManager();
            
            // add text fields and position them
            var hdt:TextField = new TextField();
            hdt.text = "Byte Converter";
            hdt.x = 20;
            hdt.y = -150;
            hdt.width = 220;
            hdt.mouseEnabled = false;
            hdt.setTextFormat(_hf);
            _ksd.addChild(hdt);
            
            _kilobytes = new TextField();
            _kilobytes.mouseEnabled = false;
            _kilobytes.text = "Kilobytes: 0";
            _ksd.addChild(_kilobytes);
            _kilobytes.setTextFormat(_tf);
            _kilobytes.width = 250;
            _kilobytes.y = -120;
            _kilobytes.x = 10;
            
            _kbToMB = new TextField();
            _kbToMB.text = "Megabytes: 0";
            _kbToMB.mouseEnabled = false;
            _ksd.addChild(_kbToMB);
            _kbToMB.setTextFormat(_tf);
            _kbToMB.width = 200;
            _kbToMB.y = -95;
            _kbToMB.x = 10;
                    
            _mbToGB = new TextField();
            _mbToGB.mouseEnabled = false;
            _mbToGB.text = "Gigabytes: 0";
            _ksd.addChild(_mbToGB);
            _mbToGB.setTextFormat(_tf);
            _mbToGB.width = 200;
            _mbToGB.y = -70;
            _mbToGB.x = 10;
                
            _gbToTB = new TextField();
            _gbToTB.mouseEnabled = false;
            _gbToTB.text = "Terabytes: 0";
            _ksd.addChild(_gbToTB);
            _gbToTB.setTextFormat(_tf);
            _gbToTB.width = 200;
            _gbToTB.y = -45;
            _gbToTB.x = 10;
            
            // event listener for handle position changes
            _ksm.setUpAssets(_ksd.mcTrack, _ksd.mcHandle);
            _ksm.addEventListener(Event.CHANGE, onChangeHandler);
            
        }
        // function that converts the byte values by calling the static utility function
        private function onChangeHandler(event:Event):void
        {
            if (_ksm.percent > 0.997)
            {
                _ksm.percent = 1;
            }
            
            // static utility functions at work on the value passed from the slider
            _kilobytes.text = "Kilobytes: " + KilobyteConverterUtil.kb(_ksm.percent);
            _kbToMB.text = "Megabytes: " + KilobyteConverterUtil.kbToMB(_ksm.percent);
            _mbToGB.text = "Gigabytes: " +  KilobyteConverterUtil.mbToGB(_ksm.percent) ;
            _gbToTB.text = "Terabytes: " + KilobyteConverterUtil.gbToTB(_ksm.percent);

            //format the text after it has been altered
            _kilobytes.setTextFormat(_tf);
            _kbToMB.setTextFormat(_tf);
            _mbToGB.setTextFormat(_tf);
            _gbToTB.setTextFormat(_tf);
            
        }
        
        // a function that converts inches to feet

        private function inchesToFeetConverter():void
        {
            //add slider asset to stage 
            _sd = new SliderDisplay();
            _sd.scaleX = _sd.scaleY = .75;
            _sd.x = 75;
            _sd.y = 250;
            
            // set text fields with values and position
            var hdt:TextField = new TextField();
            hdt.text = "Inches to Feet"
            hdt.x = 25;
            hdt.y = -150;
            hdt.width = 200;
            hdt.mouseEnabled = false;
            hdt.setTextFormat(_hf);
            
            _sd.addChild(hdt);
            _sdInches = new TextField();
            _sdFeet = new TextField();
            _sdInches.text = "INCHES: 0";
            _sdFeet.text = "FEET: 0";
            _sdInches.setTextFormat(_tf);
            _sdFeet.setTextFormat(_tf);
            _sdInches.mouseEnabled = false;
            _sdFeet.mouseEnabled = false;
            
            _sd.addChild(_sdInches);
            _sd.addChild(_sdFeet);
            _sdFeet.width = 200;
            _sdInches.width = 200;
            _sdInches.y = -100;
            _sdFeet.y = -80;
            _sdFeet.x = _sdInches.x = 25;
            
            //add to stage in layoutbox
            _lb.addChild(_sd);
            
            // set up the slide manager and the event listener
            // for the position of handle changes
            _sdm = new SlideManager();
            _sdm.setUpAssets(_sd.mcTrack, _sd.mcHandle);
            _sdm.addEventListener(Event.CHANGE, onChange);

        }
            
        // static function that converts inches to feet 
        // and updates the text fields
        
        private function onChange(event:Event):void
        {
            _sm = SlideManager(event.currentTarget);
            _sm.percent = _sdm.percent;
                if(_sm.percent >= 0.995)
                    {
                        _sm.percent = Math.ceil(_sm.percent);    
                    }
            
            _sdInches.text = "INCHES: " + InchesToFeet.convertInches(_sm.percent);
            _sdFeet.text = "FEET: " + InchesToFeet.convertFeet(_sm.percent).toFixed(2);    
            
            //format text after updating
            _sdInches.setTextFormat(_tf);
            _sdFeet.setTextFormat(_tf);
            
        }
    }
}
