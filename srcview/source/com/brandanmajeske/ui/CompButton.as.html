package com.brandanmajeske.ui
{
    import flash.display.Graphics;
    import flash.display.Sprite;
    
    public class CompButton extends Sprite
    {
        
        private var _width:int;
        private var _height:int;
        private var _label:Label;
        private var _backgroundColor:uint = 0xffbb00;
        private var _borderColor:uint = 0x555555;
        private var _labelColor:uint = 0x222222;
        
        public function CompButton(width:int = 100, height:int = 40)
        {
            super();
            _width = width;
            _height = height;
            createLabel();
            draw();
            alignLabel();            
            
        }
        
        private function createLabel():void
        {
            _label = new Label();
            _label.color = _labelColor;
            _label.text = "Testing...";
            this.addChild(_label);
        }
            
        private function draw():void
        {
            var g:Graphics = this.graphics;
            g.clear();
            g.beginFill( _backgroundColor, 1);
            g.lineStyle(1,_borderColor, 1, true);
            g.drawRoundRect(0,0, _width, _height, 6,6);
            g.endFill();
        }
        
        private function alignLabel():void
        {
            _label.x = _width/2 - _label.width /2;
            _label.y = _height/2 - _label.height /2;
        }
        

        // ----------------------------- GETTER / SETTER ---------------------------//


        override public function get width():Number
        {
            return _width;
        }

        override public function set width(value:Number):void
        {
            _width = value;
            draw();
            alignLabel();
        }

        override public function get height():Number
        {
            return _height;
        }

        override public function set height(value:Number):void
        {
            _height = value;
            draw();
            alignLabel();
        }

        public function get label():String
        {
            return _label.text;
        }

        public function set label(value:String):void
        {
            _label.text = value;
            alignLabel();
        }

        public function get backgroundColor():uint
        {
            return _backgroundColor;
        }

        public function set backgroundColor(value:uint):void
        {
            _backgroundColor = value;
            draw();
        }

        public function get borderColor():uint
        {
            return _borderColor;
        }

        public function set borderColor(value:uint):void
        {
            _borderColor = value;
            draw();
        }

        public function get labelColor():uint
        {
            return _labelColor;
        }

        public function set labelColor(value:uint):void
        {
            _labelColor = value;
            _label.color = _labelColor;
            draw();
        }


    }
}
